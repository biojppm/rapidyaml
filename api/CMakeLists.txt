function(create_setup_py src dst)
    set(opt0arg
    )
    set(opt1arg
        NAME
        VERSION
        AUTHOR
        DESC
        SRC
    )
    set(optnarg
        PYMODS
    )
    cmake_parse_arguments("" "${opt0arg}" "${opt1arg}" "${optnarg}" "${ARGN}")
    configure_file("${src}" "${dst}" @ONLY)
endfunction()

function(pylist li out)
    print_var(li)
    set(l)
    foreach(f ${li})
        set(l "${l}'${f}', ")
    endforeach()
    set(${out} ${l} PARENT_SCOPE)
endfunction()
